usage

1. Include all necessary .css and .js files in <HEAD> area:

                
<!--All-->
<script src="./jQuery/jquery-1.12.4.min.js"></script> 
<!--PDF--> <script src="./pdf/pdf.js"></script> 
<!--Docs--> <script src="./docxjs/jszip-utils.js"></script> 
<script src="./docxjs/mammoth.browser.min.js"></script> 
<!--PPTX--> 
<link rel="stylesheet" href="./pptxjs/pptxjs.css"> 
<link rel="stylesheet" href="./pptxjs/nv.d3.min.css"> 
<script type="text/javascript" src="./pptxjs/jszip.min.js"></script> 
<script type="text/javascript" src="./pptxjs/filereader.js"></script> 
<script type="text/javascript" src="./pptxjs/d3.min.js"></script> 
<script type="text/javascript" src="./pptxjs/nv.d3.min.js"></script> 
<script type="text/javascript" src="./pptxjs/pptxjs.js"></script>
<script type="text/javascript" src="./pptxjs/divs2slides.js"></script> 
<!--All Spreadsheet --> 
<script type="text/javascript" src="./sheetjs/jquery.handsontable.full.js"></script> 
<script type="text/javascript" src="./sheetjs/xlsx.full.min.js"></script> 
<link rel="stylesheet" href="./sheetjs/jquery.handsontable.full.css"> 
<!--All--> 
<script src="officejs.js"></script>

2. Add <Div> with id in <BODY> area:

                
<div id="resolte-contaniner"></div>

3.Use one or both javascript method to read file:

a. For reading remote file :

                
var file_path = "path/to/file.pptx"; 
$("#resolte-contaniner").officejs({
   url: file_path
});

a. For reading local file using INPUT object :

a1. First add <Input> with id in <BODY> area:

                
<input type="file" id="select_file" />

a2. And then use next script:

                
$("#resolte-contaniner").officejs({
  inputObjId: "select_file"
});
